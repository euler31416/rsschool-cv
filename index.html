Number of lower and upper chars

            
                #include stdio.h
                    #include ctype.h
                    
                    #define INPUT 100			
                    
                    int main(void)
                    {
                        int input[INPUT];
                        int i = 0;
                        int low = 0;
                        int up = 0;
                        int other = 0;
                        
                        while((input[i++] = getchar()) != '\n');
                        
                        for(i = 0; input[i] != '\n'; i++)
                        {
                            if (islower(input[i])){
                                low++;
                            } else if(isupper(input[i])){
                                up++;
                            } else other++;
                            
                        }
                        printf("We have %d upper chars, %d lower chars and %d other chars.\n", up, low, other);
                        
                        return 0;
                    }
